1. const express = require('express');

2. const app = express()

3. const port = 5000;

const users = [
{id : 0,name:"bipul chandro roy",age:26},
{id : 1,name:"sagar chandro roy",age:23},

{id : 3,name:"mithun chandro roy",age:23},

]

4. app.get('/users',(req,res)=>{
res.send(users)
})

5. app.get('/users/:id',(req,res)=>{
const id = req.param.id
const user = users[id]
res.send(user) })

6.app.listen(port,()=>{ console.log('listening the port',port) })

Dynamic routing by query

1. const express = require('express');

2. const app = express()

3. const port = 5000;

const users = [
{id : 0,name:"bipul chandro roy",age:26},
{id : 1,name:"sagar chandro roy",age:23},
{id : 2,name:"sujon chandro roy",age:25},
{id : 3,name:"mithun chandro roy",age:23},
{id : 4,name:"uttam chandro roy",age:25}
]

4. app.get('/users',(req,res)=>{
const search = req.query.search
if(search){
const searchResult = users.filter(user => user.name.toLocaleLowerCase().includes(search))
res.send(searchResult)
}else{
res.send(users)
} })

5.app.listen(port,()=>{ console.log('listening the port',port) })

Fetch to load data, handle cors

  • first install cors by giving this command in command propmt npm install cors
  • then go to call it in top under express call

it will give you a to not write again and again node index.js.and now go to your command prompt and type npm run start-dev

1. const express = require('express');

2. const cors = require('cors')

3. const app = express()

4. app.use(cors())

5. app.use(express.json())

6. const port = 5000;

const users = [
{id : 0,name:"bipul chandro roy",age:26},
{id : 1,name:"sagar chandro roy",age:23},
{id : 2,name:"sujon chandro roy",age:25},
{id : 3,name:"mithun chandro roy",age:23},
{id : 4,name:"uttam chandro roy",age:25}
]

7. app.get('/users',(req,res)=>{
res.send(users)
})

8.app.post('/users',(req,res)=>{
const newUser = req.body
newUser.id = users.length
users.push(newUser)
res.json(newUser)
})

app.listen(port,()=>{
console.log('listening the port',port)
})

to set json format...yo have to call app.use(express.json())in top
